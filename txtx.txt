import React, { useState } from 'react';
import { Card, CardContent } from '@/components/ui/card';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import {
  Select,
  SelectTrigger,
  SelectValue,
  SelectContent,
  SelectItem,
} from '@/components/ui/select';
import { Button } from '@/components/ui/button';
import { Download, RotateCcw, Share2 } from 'lucide-react';

const GenerateQR = () => {
  const [formData, setFormData] = useState({
    courseTitle: '',
    courseCode: '',
    level: '',
    duration: ''
  });
  const [isGenerating, setIsGenerating] = useState(false);
  const [qrGenerated, setQrGenerated] = useState(false);
  const [qrCodeUrl, setQrCodeUrl] = useState('');

  const handleInputChange = (field, value) => {
    setFormData(prev => ({
      ...prev,
      [field]: value
    }));
  };

  const handleGenerate = async () => {
    if (!formData.courseTitle || !formData.courseCode || !formData.level || !formData.duration) {
      alert('Please fill in all fields');
      return;
    }

    setIsGenerating(true);
    
    try {
      // Simulate API call to backend
      await new Promise(resolve => setTimeout(resolve, 2000));
      
      // Mock QR code URL - replace with actual backend response
      setQrCodeUrl('/api/placeholder/200/200');
      setQrGenerated(true);
    } catch (error) {
      console.error('Error generating QR code:', error);
      alert('Failed to generate QR code. Please try again.');
    } finally {
      setIsGenerating(false);
    }
  };

  const handleCancel = () => {
    setFormData({
      courseTitle: '',
      courseCode: '',
      level: '',
      duration: ''
    });
    setQrGenerated(false);
    setQrCodeUrl('');
    setIsGenerating(false);
  };

  const handleReset = () => {
    setQrGenerated(false);
    setQrCodeUrl('');
  };

  const handleDownload = () => {
    // Simulate download functionality
    const link = document.createElement('a');
    link.href = qrCodeUrl;
    link.download = `${formData.courseCode}-qr-code.png`;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  const QRPlaceholder = ({ processed = false, generated = false }) => (
    <div className="bg-white rounded-lg p-8 flex flex-col items-center justify-center min-h-64 border border-gray-200">
      {generated ? (
        <img 
          src={qrCodeUrl} 
          alt="Generated QR Code" 
          className="w-32 h-32 mb-4"
        />
      ) : (
        <div className="w-20 h-20 mb-4 flex items-center justify-center">
          <div className="grid grid-cols-2 gap-1">
            <div className="w-4 h-4 bg-gray-800 rounded-sm"></div>
            <div className="w-4 h-4 bg-white border border-gray-800 rounded-sm"></div>
            <div className="w-4 h-4 bg-white border border-gray-800 rounded-sm"></div>
            <div className="w-4 h-4 bg-gray-800 rounded-sm"></div>
          </div>
        </div>
      )}
      <p className="text-gray-500 text-sm text-center">
        {generated ? 'QR Code has been generated' : 'QR Code not processed yet'}
      </p>
    </div>
  );

  return (
    <div className="container mx-auto p-6 space-y-6">
      <div>
        <h2 className="font-semibold text-xl md:text-2xl text-gray-800">
          Generate QR Code for Attendance
        </h2>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
        {/* Left Side - Form */}
        <Card className="bg-white border border-gray-200 shadow-sm">
          <CardContent className="p-6">
            <div className="space-y-6">
              <div className="space-y-2">
                <Label htmlFor="courseTitle" className="text-sm font-medium text-gray-700">
                  Course Title
                </Label>
                <Input
                  type="text"
                  id="courseTitle"
                  value={formData.courseTitle}
                  onChange={(e) => handleInputChange('courseTitle', e.target.value)}
                  placeholder="INTRODUCTION TO PYTHON"
                  className="w-full h-12 bg-gray-50 border border-gray-200 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-all"
                />
              </div>

              <div className="grid grid-cols-2 gap-4">
                <div className="space-y-2">
                  <Label htmlFor="courseCode" className="text-sm font-medium text-gray-700">
                    Course Code
                  </Label>
                  <Input
                    type="text"
                    id="courseCode"
                    value={formData.courseCode}
                    onChange={(e) => handleInputChange('courseCode', e.target.value)}
                    placeholder="CSC 309"
                    className="w-full h-12 bg-gray-50 border border-gray-200 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-all"
                  />
                </div>

                <div className="space-y-2">
                  <Label htmlFor="time" className="text-sm font-medium text-gray-700">
                    QR Code Duration
                  </Label>
                  <div className="flex">
                    <Input
                      type="number"
                      value={formData.duration}
                      onChange={(e) => handleInputChange('duration', e.target.value)}
                      placeholder="30"
                      className="w-20 h-12 bg-gray-50 border border-gray-200 rounded-l-md focus:ring-2 focus:ring-blue-500 focus:border-blue-500 transition-all"
                    />
                    <div className="bg-gray-100 border-t border-b border-r border-gray-200 rounded-r-md px-3 flex items-center">
                      <span className="text-sm text-gray-600">mins</span>
                    </div>
                  </div>
                </div>
              </div>

              <div className="space-y-2">
                <Label htmlFor="level" className="text-sm font-medium text-gray-700">
                  Level
                </Label>
                <Select 
                  value={formData.level} 
                  onValueChange={(value) => handleInputChange('level', value)}
                >
                  <SelectTrigger className="h-12 bg-gray-50 border border-gray-200 rounded-md focus:ring-2 focus:ring-blue-500 focus:border-blue-500">
                    <SelectValue placeholder="Select course level" />
                  </SelectTrigger>
                  <SelectContent>
                    <SelectItem value="100">100 Level</SelectItem>
                    <SelectItem value="200">200 Level</SelectItem>
                    <SelectItem value="300">300 Level</SelectItem>
                    <SelectItem value="400">400 Level</SelectItem>
                    <SelectItem value="500">500 Level</SelectItem>
                  </SelectContent>
                </Select>
              </div>

              {/* Before Generation State */}
              {!qrGenerated && (
                <>
                  <QRPlaceholder />
                  
                  <div className="flex gap-4">
                    <Button 
                      onClick={handleGenerate}
                      disabled={isGenerating}
                      className="flex-1 bg-teal-600 hover:bg-teal-700 text-white h-12 rounded-md font-medium transition-colors"
                    >
                      {isGenerating ? 'Generating...' : 'Generate'}
                    </Button>
                    <Button 
                      onClick={handleCancel}
                      variant="outline"
                      className="flex-1 border-red-500 text-red-500 hover:bg-red-50 h-12 rounded-md font-medium transition-colors"
                    >
                      Cancel
                    </Button>
                  </div>

                  <p className="text-center text-sm text-gray-500 mt-4">
                    Simply generate the QR Code for the lecture
                  </p>
                </>
              )}
            </div>
          </CardContent>
        </Card>

        {/* Right Side - QR Code Display (After Generation) */}
        {qrGenerated && (
          <Card className="bg-white border border-gray-200 shadow-sm">
            <CardContent className="p-6">
              <div className="space-y-4">
                {/* Course Info Header */}
                <div className="text-center space-y-1">
                  <h3 className="font-semibold text-lg text-gray-800">
                    {formData.courseTitle}
                  </h3>
                  <div className="flex justify-center items-center gap-4 text-sm text-gray-600">
                    <span>{formData.courseCode}</span>
                    <span>•</span>
                    <span>11:30AM</span>
                    <span>•</span>
                    <span>{formData.duration} mins</span>
                  </div>
                </div>

                {/* QR Code */}
                <QRPlaceholder generated={true} />

                <p className="text-center text-sm text-gray-600">
                  QR Code has been generated
                </p>

                <p className="text-center text-sm text-red-500">
                  Expires in 29mins 28sec
                </p>

                {/* Action Buttons */}
                <div className="space-y-3">
                  <Button 
                    onClick={handleDownload}
                    className="w-full bg-teal-600 hover:bg-teal-700 text-white h-12 rounded-md font-medium transition-colors flex items-center justify-center gap-2"
                  >
                    <Download size={18} />
                    Download QR Code
                  </Button>

                  <div className="flex gap-3">
                    <Button 
                      onClick={handleReset}
                      variant="outline"
                      className="flex-1 border-teal-600 text-teal-600 hover:bg-teal-50 h-10 rounded-md transition-colors flex items-center justify-center gap-2"
                    >
                      <RotateCcw size={16} />
                      Reset
                    </Button>
                    <Button 
                      variant="outline"
                      className="border-teal-600 text-teal-600 hover:bg-teal-50 h-10 px-4 rounded-md transition-colors"
                    >
                      <Share2 size={16} />
                    </Button>
                  </div>
                </div>
              </div>
            </CardContent>
          </Card>
        )}
      </div>
    </div>
  );
};

export default GenerateQR;
